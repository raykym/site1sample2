412=head2 query
413
414 my $query = $url->query;
415 $url = $url->query([merge => 'with']);
416 $url = $url->query({append => 'to'});
417 $url = $url->query(replace => 'with');
418 $url = $url->query('a=1&b=2');
419 $url = $url->query(Mojo::Parameters->new);
420
421Query part of this URL, key/value pairs in an array reference will be merged
422with L<Mojo::Parameters/"merge">, and key/value pairs in a hash reference
423appended with L<Mojo::Parameters/"append">, defaults to a L<Mojo::Parameters>
424object.
425
426 # "2"
427 Mojo::URL->new('http://example.com?a=1&b=2')->query->param('b');
428
429 # "a=2&b=2&c=3"
430 Mojo::URL->new('http://example.com?a=1&b=2')->query->merge(a => 2, c => 3);
431
432 # "http://example.com?a=2&c=3"
433 Mojo::URL->new('http://example.com?a=1&b=2')->query(a => 2, c => 3);
434
435 # "http://example.com?a=2&a=3"
436 Mojo::URL->new('http://example.com?a=1&b=2')->query(a => [2, 3]);
437
438 # "http://example.com?a=2&b=2&c=3"
439 Mojo::URL->new('http://example.com?a=1&b=2')->query([a => 2, c => 3]);
440
441 # "http://example.com?b=2"
442 Mojo::URL->new('http://example.com?a=1&b=2')->query([a => undef]);
443
444 # "http://example.com?a=1&b=2&a=2&c=3"
445 Mojo::URL->new('http://example.com?a=1&b=2')->query({a => 2, c => 3});
