73=head1 NAME
74
75Mojo::Log - Simple logger
76
77=head1 SYNOPSIS
78
79 use Mojo::Log;
80
81 # Log to STDERR
82 my $log = Mojo::Log->new;
83
84 # Customize log file location and minimum log level
85 my $log = Mojo::Log->new(path => '/var/log/mojo.log', level => 'warn');
86
87 # Log messages
88 $log->debug('Not sure what is happening here');
89 $log->info('FYI: it happened again');
90 $log->warn('This might be a problem');
91 $log->error('Garden variety error');
92 $log->fatal('Boom');
93
94=head1 DESCRIPTION
95
96L<Mojo::Log> is a simple logger for L<Mojo> projects.
97
98=head1 EVENTS
99
100L<Mojo::Log> inherits all events from L<Mojo::EventEmitter> and can emit the
101following new ones.
102
103=head2 message
104
105 $log->on(message => sub {
106 my ($log, $level, @lines) = @_;
107 ...
108 });
109
110Emitted when a new message gets logged.
111
112 $log->unsubscribe('message')->on(message => sub {
113 my ($log, $level, @lines) = @_;
114 say "$level: ", @lines;
115 });
116
117=head1 ATTRIBUTES
118
119L<Mojo::Log> implements the following attributes.
120
121=head2 format
122
123 my $cb = $log->format;
124 $log = $log->format(sub {...});
125
126A callback for formatting log messages.
127
128 $log->format(sub {
129 my ($time, $level, @lines) = @_;
130 return "[Thu May 15 17:47:04 2014] [info] I ♥ Mojolicious\n";
131 });
132
133=head2 handle
134
135 my $handle = $log->handle;
136 $log = $log->handle(IO::Handle->new);
137
138Log filehandle used by default L</"message"> event, defaults to opening
139L</"path"> or C<STDERR>.
140
141=head2 history
142
143 my $history = $log->history;
144 $log = $log->history([[time, 'debug', 'That went wrong']]);
145
146The last few logged messages.
147
148=head2 level
149
150 my $level = $log->level;
151 $log = $log->level('debug');
152
153Active log level, defaults to C<debug>. Available log levels are C<debug>,
154C<info>, C<warn>, C<error> and C<fatal>, in that order. Note that the
155C<MOJO_LOG_LEVEL> environment variable can override this value.
156
157=head2 max_history_size
158
159 my $size = $log->max_history_size;
160 $log = $log->max_history_size(5);
161
162Maximum number of logged messages to store in L</"history">, defaults to C<10>.
163
164=head2 path
165
166 my $path = $log->path
167 $log = $log->path('/var/log/mojo.log');
168
169Log file path used by L</"handle">.
170
171=head1 METHODS
172
173L<Mojo::Log> inherits all methods from L<Mojo::EventEmitter> and implements the
174following new ones.
175
176=head2 append
177
178 $log->append("[Thu May 15 17:47:04 2014] [info] I ♥ Mojolicious\n");
179
180Append message to L</"handle">.
181
182=head2 debug
183
184 $log = $log->debug('You screwed up, but that is ok');
185 $log = $log->debug('All', 'cool');
186
187Emit L</"message"> event and log C<debug> message.
188
189=head2 error
190
191 $log = $log->error('You really screwed up this time');
192 $log = $log->error('Wow', 'seriously');
193
194Emit L</"message"> event and log C<error> message.
195
196=head2 fatal
197
198 $log = $log->fatal('Its over...');
199 $log = $log->fatal('Bye', 'bye');
200
201Emit L</"message"> event and log C<fatal> message.
202
203=head2 info
204
205 $log = $log->info('You are bad, but you prolly know already');
206 $log = $log->info('Ok', 'then');
207
208Emit L</"message"> event and log C<info> message.
209
210=head2 is_level
211
212 my $bool = $log->is_level('debug');
213
214Check active log L</"level">.
215
216 # True
217 $log->level('debug')->is_level('debug');
218 $log->level('debug')->is_level('info');
219
220 # False
221 $log->level('info')->is_level('debug');
222 $log->level('fatal')->is_level('warn');
223
224=head2 new
225
226 my $log = Mojo::Log->new;
227
228Construct a new L<Mojo::Log> object and subscribe to L</"message"> event with
229default logger.
230
231=head2 warn
232
233 $log = $log->warn('Dont do that Dave...');
234 $log = $log->warn('No', 'really');
235
236Emit L</"message"> event and log C<warn> message.
