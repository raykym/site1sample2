150=head1 NAME
151
152Mojolicious::Static - Serve static files
153
154=head1 SYNOPSIS
155
156 use Mojolicious::Static;
157
158 my $static = Mojolicious::Static->new;
159 push @{$static->classes}, 'MyApp::Controller::Foo';
160 push @{$static->paths}, '/home/sri/public';
161
162=head1 DESCRIPTION
163
164L<Mojolicious::Static> is a static file server with C<Range>,
165C<If-Modified-Since> and C<If-None-Match> support, based on
166L<RFC 7232|http://tools.ietf.org/html/rfc7232> and
167L<RFC 7233|http://tools.ietf.org/html/rfc7233>.
168
169=head1 ATTRIBUTES
170
171L<Mojolicious::Static> implements the following attributes.
172
173=head2 classes
174
175 my $classes = $static->classes;
176 $static = $static->classes(['main']);
177
178Classes to use for finding files in C<DATA> sections with L<Mojo::Loader>,
179first one has the highest precedence, defaults to C<main>. Only files with
180exactly one extension will be used, like C<index.html>. Note that for files to
181be detected, these classes need to have already been loaded and added before
182L</"warmup"> is called, which usually happens automatically during application
183startup.
184
185 # Add another class with static files in DATA section
186 push @{$static->classes}, 'Mojolicious::Plugin::Fun';
187
188 # Add another class with static files in DATA section and higher precedence
189 unshift @{$static->classes}, 'Mojolicious::Plugin::MoreFun';
190
191=head2 paths
192
193 my $paths = $static->paths;
194 $static = $static->paths(['/home/sri/public']);
195
196Directories to serve static files from, first one has the highest precedence.
197
198 # Add another "public" directory
199 push @{$static->paths}, '/home/sri/public';
200
201 # Add another "public" directory with higher precedence
202 unshift @{$static->paths}, '/home/sri/themes/blue/public';
203
204=head1 METHODS
205
206L<Mojolicious::Static> inherits all methods from L<Mojo::Base> and implements
207the following new ones.
208
209=head2 dispatch
210
211 my $bool = $static->dispatch(Mojolicious::Controller->new);
212
213Serve static file for L<Mojolicious::Controller> object.
214
215=head2 file
216
217 my $asset = $static->file('images/logo.png');
218 my $asset = $static->file('../lib/MyApp.pm');
219
220Build L<Mojo::Asset::File> or L<Mojo::Asset::Memory> object for a file,
221relative to L</"paths"> or from L</"classes">, or return C<undef> if it doesn't
222exist. Note that this method uses a relative path, but does not protect from
223traversing to parent directories.
224
225 my $content = $static->file('foo/bar.html')->slurp;
226
227=head2 is_fresh
228
229 my $bool = $static->is_fresh(Mojolicious::Controller->new, {etag => 'abc'});
230
231Check freshness of request by comparing the C<If-None-Match> and
232C<If-Modified-Since> request headers to the C<ETag> and C<Last-Modified>
233response headers.
234
235These options are currently available:
236
237=over 2
238
239=item etag
240
241 etag => 'abc'
242
243Add C<ETag> header before comparing.
244
245=item last_modified
246
247 last_modified => $epoch
248
249Add C<Last-Modified> header before comparing.
250
251=back
252
253=head2 serve
254
255 my $bool = $static->serve(Mojolicious::Controller->new, 'images/logo.png');
256 my $bool = $static->serve(Mojolicious::Controller->new, '../lib/MyApp.pm');
257
258Serve a specific file, relative to L</"paths"> or from L</"classes">. Note that
259this method uses a relative path, but does not protect from traversing to parent
260directories.
261
262=head2 serve_asset
263
264 $static->serve_asset(Mojolicious::Controller->new, Mojo::Asset::File->new);
265
266Serve a L<Mojo::Asset::File> or L<Mojo::Asset::Memory> object with C<Range>,
267C<If-Modified-Since> and C<If-None-Match> support.
268
269=head2 warmup
270
271 $static->warmup;
272
273Prepare static files from L</"classes"> for future use.
274
