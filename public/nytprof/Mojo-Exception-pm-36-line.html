92=head1 NAME
93
94Mojo::Exception - Exceptions with context
95
96=head1 SYNOPSIS
97
98 use Mojo::Exception;
99
100 # Throw exception and show stack trace
101 eval { Mojo::Exception->throw('Something went wrong!') };
102 say "$_->[1]:$_->[2]" for @{$@->frames};
103
104 # Customize exception
105 eval {
106 my $e = Mojo::Exception->new('Died at test.pl line 3.');
107 die $e->trace(2)->inspect->verbose(1);
108 };
109 say $@;
110
111=head1 DESCRIPTION
112
113L<Mojo::Exception> is a container for exceptions with context information.
114
115=head1 ATTRIBUTES
116
117L<Mojo::Exception> implements the following attributes.
118
119=head2 frames
120
121 my $frames = $e->frames;
122 $e = $e->frames([$frame1, $frame2]);
123
124Stack trace if available.
125
126 # Extract information from the last frame
127 my ($package, $filename, $line, $subroutine, $hasargs, $wantarray, $evaltext,
128 $is_require, $hints, $bitmask, $hinthash) = @{$e->frames->[-1]};
129
130=head2 line
131
132 my $line = $e->line;
133 $e = $e->line([3, 'die;']);
134
135The line where the exception occurred if available.
136
137=head2 lines_after
138
139 my $lines = $e->lines_after;
140 $e = $e->lines_after([[4, 'say $foo;'], [5, 'say $bar;']]);
141
142Lines after the line where the exception occurred if available.
143
144=head2 lines_before
145
146 my $lines = $e->lines_before;
147 $e = $e->lines_before([[1, 'my $foo = 23;'], [2, 'my $bar = 24;']]);
148
149Lines before the line where the exception occurred if available.
150
151=head2 message
152
153 my $msg = $e->message;
154 $e = $e->message('Died at test.pl line 3.');
155
156Exception message, defaults to C<Exception!>.
157
158=head2 verbose
159
160 my $bool = $e->verbose;
161 $e = $e->verbose($bool);
162
163Enable context information for L</"to_string">.
164
165=head1 METHODS
166
167L<Mojo::Exception> inherits all methods from L<Mojo::Base> and implements the
168following new ones.
169
170=head2 inspect
171
172 $e = $e->inspect;
173 $e = $e->inspect($source1, $source2);
174
175Inspect L</"message">, L</"frames"> and optional additional sources to fill
176L</"lines_before">, L</"line"> and L</"lines_after"> with context information.
177
178=head2 new
179
180 my $e = Mojo::Exception->new;
181 my $e = Mojo::Exception->new('Died at test.pl line 3.');
182
183Construct a new L<Mojo::Exception> object and assign L</"message"> if necessary.
184
185=head2 to_string
186
187 my $str = $e->to_string;
188
189Render exception.
190
191 # Render exception with context
192 say $e->verbose(1)->to_string;
193
194=head2 throw
195
196 Mojo::Exception->throw('Something went wrong!');
197
198Throw exception from the current execution context.
199
200 # Longer version
201 die Mojo::Exception->new('Something went wrong!')->trace->inspect;
202
203=head2 trace
204
205 $e = $e->trace;
206 $e = $e->trace($skip);
207
208Generate stack trace and store all L</"frames">, defaults to skipping C<1> call
209frame.
210
211 # Skip 3 call frames
212 $e->trace(3);
213
214 # Skip no call frames
215 $e->trace(0);
216
217=head1 OPERATORS
218
219L<Mojo::Exception> overloads the following operators.
220
221=head2 bool
222
223 my $bool = !!$e;
224
225Always true.
226
227=head2 stringify
