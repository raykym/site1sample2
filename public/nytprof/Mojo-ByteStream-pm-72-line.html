69=head1 NAME
70
71Mojo::ByteStream - ByteStream
72
73=head1 SYNOPSIS
74
75 use Mojo::ByteStream;
76
77 # Manipulate bytestream
78 my $stream = Mojo::ByteStream->new('foo_bar_baz');
79 say $stream->camelize;
80
81 # Chain methods
82 my $stream = Mojo::ByteStream->new('foo bar baz')->quote;
83 $stream = $stream->unquote->encode('UTF-8')->b64_encode('');
84 say "$stream";
85
86 # Use the alternative constructor
87 use Mojo::ByteStream 'b';
88 my $stream = b('foobarbaz')->b64_encode('')->say;
89
90=head1 DESCRIPTION
91
92L<Mojo::ByteStream> is a scalar-based container for bytestreams that provides a
93more friendly API for many of the functions in L<Mojo::Util>.
94
95 # Access scalar directly to manipulate bytestream
96 my $stream = Mojo::ByteStream->new('foo');
97 $$stream .= 'bar';
98
