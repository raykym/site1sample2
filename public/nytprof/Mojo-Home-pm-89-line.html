60=head1 NAME
61
62Mojo::Home - Home sweet home
63
64=head1 SYNOPSIS
65
66 use Mojo::Home;
67
68 # Find and manage the project root directory
69 my $home = Mojo::Home->new;
70 $home->detect;
71 say $home->lib_dir;
72 say $home->rel_file('templates/layouts/default.html.ep');
73 say "$home";
74
75=head1 DESCRIPTION
76
77L<Mojo::Home> is a container for home directories.
78
79=head1 ATTRIBUTES
80
81L<Mojo::Home> implements the following attributes.
82
83=head2 parts
84
85 my $parts = $home->parts;
86 $home = $home->parts(['home', 'sri', 'myapp']);
87
88Home directory parts.
89
90=head1 METHODS
91
92L<Mojo::Home> inherits all methods from L<Mojo::Base> and implements the
93following new ones.
94
95=head2 detect
96
97 $home = $home->detect;
98 $home = $home->detect('My::App');
99
100Detect home directory from the value of the C<MOJO_HOME> environment variable,
101location of the application class, or the current working directory.
102
103=head2 lib_dir
104
105 my $path = $home->lib_dir;
106
107Path to C<lib> directory of application.
108
109=head2 list_files
110
111 my $files = $home->list_files;
112 my $files = $home->list_files('foo/bar');
113 my $files = $home->list_files('foo/bar', {hidden => 1});
114
115Portably list all files recursively in directory relative to the home directory.
116
117 # List layouts
118 say $home->rel_file($_) for @{$home->list_files('templates/layouts')};
119
120These options are currently available:
121
122=over 2
123
124=item hidden
125
126 hidden => 1
127
128Include hidden files and directories.
129
130=back
131
132=head2 mojo_lib_dir
133
134 my $path = $home->mojo_lib_dir;
135
136Path to C<lib> directory in which L<Mojolicious> is installed.
137

142

144necessary.
145
146=head2 parse
