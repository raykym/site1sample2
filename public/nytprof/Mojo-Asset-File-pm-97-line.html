136=head1 NAME
137
138Mojo::Asset::File - File storage for HTTP content
139
140=head1 SYNOPSIS
141
142 use Mojo::Asset::File;
143
144 # Temporary file
145 my $file = Mojo::Asset::File->new;
146 $file->add_chunk('foo bar baz');
147 say 'File contains "bar"' if $file->contains('bar') >= 0;
148 say $file->slurp;
149
150 # Existing file
151 my $file = Mojo::Asset::File->new(path => '/home/sri/foo.txt');
152 $file->move_to('/yada.txt');
153 say $file->slurp;
154
155=head1 DESCRIPTION
156
157L<Mojo::Asset::File> is a file storage backend for HTTP content.
158
159=head1 EVENTS
160
161L<Mojo::Asset::File> inherits all events from L<Mojo::Asset>.
162
163=head1 ATTRIBUTES
164
165L<Mojo::Asset::File> inherits all attributes from L<Mojo::Asset> and implements
166the following new ones.
167
168=head2 cleanup
169
170 my $bool = $file->cleanup;
171 $file = $file->cleanup($bool);
172
173Delete L</"path"> automatically once the file is not used anymore.
174
175=head2 handle
176
177 my $handle = $file->handle;
178 $file = $file->handle(IO::File->new);
179
180Filehandle, created on demand for L</"path">, which can be generated
181automatically and safely based on L</"tmpdir">.
182
183=head2 path
184
185 my $path = $file->path;
186 $file = $file->path('/home/sri/foo.txt');
187
188File path used to create L</"handle">.
189
190=head2 tmpdir
191
192 my $tmpdir = $file->tmpdir;
193 $file = $file->tmpdir('/tmp');
194
195Temporary directory used to generate L</"path">, defaults to the value of the
196C<MOJO_TMPDIR> environment variable or auto-detection.
197
198=head1 METHODS
199
200L<Mojo::Asset::File> inherits all methods from L<Mojo::Asset> and implements
201the following new ones.
202
203=head2 add_chunk

206
207Add chunk of data.
208
209=head2 contains

215=head2 get_chunk

222
223=head2 is_file

228
229=head2 move_to
