108=head1 NAME
109
110Mojo::Redis2::Cursor - Cursor iterator for SCAN commands.
111
112=head1 SYNOPSIS
113
114 use Mojo::Redis2;
115 use Mojo::Redis2::Cursor;
116
117 my $redis = Mojo::Redis2->new();
118 my $cursor
119 = Mojo::Redis2::Cursor->new('SCAN', match => 'mynamespace*')->redis($redis);
120
121 # blocking
122 while (my $r = $cursor->next) { say join "\n", @$r }
123
124 # or non-blocking
125 use feature 'current_sub';
126 $cursor->next(
127 sub {
128 my ($cur, $err, $r) = @_;
129 say join "\n", @$r;
130 return Mojo::IOLoop->stop() unless $cur->next(__SUB__);
131 }
132 );
133 Mojo::IOLoop->start();
134
135
136=head1 DESCRIPTION
137
138L<Mojo::Redis2::Cursor> is an iterator object for C<SCAN> family commands.
139

147information.

223=head2 next
224
225 # blocking
226 my $res = $cursor->next();
227
228 # non-blocking
229 $cursor->next(sub {
230 my ($cur, $err, $res) = @_;
231 ...
232 })
233
234Issue next C<SCAN> family command with cursor value from previous iteration. If
235last argument is coderef, will made a non-blocking call. In blocking mode returns
236arrayref with fetched elements. If no more items available, will return
237C<undef>, for both blocking and non-blocking, without calling callback.
238
239 my $res = $cursor->next(MATCH => 'namespace*');
240 $cursor->next(COUNT => 100, sub { ... });
241
242Accepts the same optional arguments as original Redis command, which will replace
243old values and will be used for this and next iterations.
244
