65=head1 NAME
66
67Mojo::EventEmitter - Event emitter base class
68
69=head1 SYNOPSIS
70
71 package Cat;
72 use Mojo::Base 'Mojo::EventEmitter';
73
74 # Emit events
75 sub poke {
76 my $self = shift;

80 package main;

90=head1 DESCRIPTION
91
92L<Mojo::EventEmitter> is a simple base class for event emitting objects.
93
94=head1 EVENTS
95
96L<Mojo::EventEmitter> can emit the following events.
97
98=head2 error
99
100 $e->on(error => sub {
101 my ($e, $err) = @_;
102 ...
103 });
104
105This is a special event for errors, it will not be emitted directly by this
106class but is fatal if unhandled.
107
108 $e->on(error => sub {
109 my ($e, $err) = @_;
110 say "This looks bad: $err";
111 });
112
113=head1 METHODS
114
115L<Mojo::EventEmitter> inherits all methods from L<Mojo::Base> and implements
116the following new ones.
117
118=head2 catch
119
120 $e = $e->catch(sub {...});
121
122Subscribe to L</"error"> event.
123
124 # Longer version
125 $e->on(error => sub {...});
126
127=head2 emit
128
129 $e = $e->emit('foo');
130 $e = $e->emit('foo', 123);
131
132Emit event.
133
134=head2 has_subscribers
135
136 my $bool = $e->has_subscribers('foo');
137
138Check if event has subscribers.
139
140=head2 on
141
142 my $cb = $e->on(foo => sub {...});
143
144Subscribe to event.
145
146 $e->on(foo => sub {
147 my ($e, @args) = @_;
148 ...
149 });
150
151=head2 once
152
153 my $cb = $e->once(foo => sub {...});
154
155Subscribe to event and unsubscribe again after it has been emitted once.
156
157 $e->once(foo => sub {
158 my ($e, @args) = @_;
159 ...
160 });
161
162=head2 subscribers
163
164 my $subscribers = $e->subscribers('foo');
165
166All subscribers for event.
167
168 # Unsubscribe last subscriber
169 $e->unsubscribe(foo => $e->subscribers('foo')->[-1]);
170
171 # Change order of subscribers
172 @{$e->subscribers('foo')} = reverse @{$e->subscribers('foo')};
173
174=head2 unsubscribe
175
176 $e = $e->unsubscribe('foo');
177 $e = $e->unsubscribe(foo => $cb);
178
179Unsubscribe from event.
